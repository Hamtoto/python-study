# OpenAI Assistants API 백엔드를 위한 nginx 설정 추가
# /etc/nginx/sites-available/your_domain.conf 파일의 server 블록 내에 추가하세요

    # --- OpenAI Assistants API 백엔드 요청 핸들링 (/openai-api로 시작하는 모든 요청) ---
    location /openai-api/ {
        proxy_pass http://localhost:8010/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

# 이 설정을 추가한 후:
# 1. sudo nginx -t 로 설정 검증
# 2. sudo systemctl reload nginx 로 nginx 리로드
# 3. https://hamtoto.kr/openai-api/ 로 접근
